function calcularMedia int() {
    var int suma;
    var int num;
    var int contador;
    suma = 0;
    num = 0;
    contador = -1;
    while (num != -1) {
        suma = suma + num;
        num = read();
        contador = contador + 1;
    }
    if (contador == 0) {
        contador = 1;
    }
    return suma / contador;
}

procedure principal() {
    var int resultado;
    resultado = calcularMedia();
    print(resultado);
}
